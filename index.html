Sub ListPdfFiles()

    Dim folderPath As String
    Dim fileName As String
    Dim rowIndex As Long
    
    folderPath = "C:\Users\yama7\test\"   ' ←ここを変更してください
    
    rowIndex = 10
    
    Range("A10:A1000").ClearContents
    
    
    fileName = Dir(folderPath & "*.pdf")
    
    If Range("A1").Value = "教育" Then
    Do While fileName <> ""
        
        ActiveSheet.Hyperlinks.Add _
            Anchor:=Cells(rowIndex, 1), _
            Address:=folderPath & fileName, _
            TextToDisplay:=fileName
        
        rowIndex = rowIndex + 1
        fileName = Dir()
        
    Loop
    
    MsgBox "PDF一覧の表示が完了しました。", vbInformation
    Else
    MsgBox ("A1が教育じゃないですよ")
End If
End Sub
